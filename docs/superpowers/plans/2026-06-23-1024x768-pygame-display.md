# 1024×768 pygame Display Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a selectable 1024×768 pygame display to PiFire that renders the dashboard/menus identically to the existing 800×480 display, with the layout rescaled to the larger 4:3 screen.

**Architecture:** The display module is resolution-agnostic (it reads all dimensions and layout from a JSON file), so the new resolution is delivered as (1) a transformed layout JSON, (2) a thin module re-exporting the existing `Display` class, and (3) a wizard manifest entry. The layout is generated deterministically from the 800×480 layout by a uniform 1.28× scale with centering ("strategy A").

**Tech Stack:** Python 3, JSON layout files, pygame (display runtime — not exercised in tests), `pytest` (dev), managed with `uv`.

## Global Constraints

- New module file: `display/dsi_1024x768t.py`, exposing a `Display` class (re-exported from `display.dsi_800x480t`).
- New layout file: `display/dsi_1024x768t.json`, generated from `display/dsi_800x480t.json`.
- Scale factor is `1024 / 800` (= 1.28); use the expression `1024 / 800` verbatim in generator and tests so float rounding matches exactly.
- Per-profile centering offsets: `profile_1` (landscape, screen 1024×768) → x_offset 0, y_offset 77; `profile_2` (portrait, screen 768×1024) → x_offset 77, y_offset 0.
- Transform per object: `position = [round(x*1024/800 + x_offset), round(y*1024/800 + y_offset)]`, `size = [round(w*1024/800), round(h*1024/800)]`. All other keys copied unchanged.
- Metadata changes only: `name` → `"dsi_1024x768t"`, `screen_width` → `1024`, `screen_height` → `768`. `splash_image`, `dash_background`, and all other metadata copied unchanged.
- No edits to `common/common.py`, `settings.json`, `display/base_flex.py`, `display/flexobject.py`, or `display/dsi_800x480t.py`. The settings default is auto-derived from the manifest by `_default_display_config()`.
- Tests run with `uv run pytest` from the repository root. Tests must NOT construct `Display` (it spawns a pygame display loop needing a framebuffer).
- Python files use TAB indentation to match the existing `display/*.py` modules. (Test files under `tests/` use 4-space indentation, matching the existing test suite.)

    out = _load(OUT)
    for profile, (xoff, yoff) in OFFSETS.items():

        for section in ('home', 'dash'):

        for section in ('menus', 'input'):

- [ ] **Step 3: Write the generator script**

Create `tools/generate_dsi_1024x768t.py`:

```python
#!/usr/bin/env python3
'''
Generate display/dsi_1024x768t.json from display/dsi_800x480t.json.

Strategy A: uniform 1.28x scale (1024/800) with centering on the slack axis.
Re-run after changing the 800x480 layout:
    python tools/generate_dsi_1024x768t.py
'''
import json
import copy
import os

SCALE = 1024 / 800  # 1.28 ; 768/480 = 1.6, so width is the binding dimension

BASE = os.path.abspath(os.path.join(os.path.dirname(__file__), '..'))
SRC = os.path.join(BASE, 'display', 'dsi_800x480t.json')
OUT = os.path.join(BASE, 'display', 'dsi_1024x768t.json')

# Rotated canvas each profile is authored for: (width, height).
TARGET_DIMS = {'profile_1': (1024, 768), 'profile_2': (768, 1024)}
SOURCE_DIMS = {'profile_1': (800, 480), 'profile_2': (480, 800)}


def _offsets(profile):
    tw, th = TARGET_DIMS[profile]
    sw, sh = SOURCE_DIMS[profile]
    return round((tw - sw * SCALE) / 2), round((th - sh * SCALE) / 2)


def _scale_obj(obj, xoff, yoff):
    if 'position' in obj:
        x, y = obj['position']
        obj['position'] = [round(x * SCALE + xoff), round(y * SCALE + yoff)]
    if 'size' in obj:
        w, h = obj['size']
        obj['size'] = [round(w * SCALE), round(h * SCALE)]


def scale_layout(src):
    data = copy.deepcopy(src)
    data['metadata']['name'] = 'dsi_1024x768t'
    data['metadata']['screen_width'] = 1024
    data['metadata']['screen_height'] = 768
    for profile in ('profile_1', 'profile_2'):
        if profile not in data:
            continue
        xoff, yoff = _offsets(profile)
        prof = data[profile]
        for section in ('home', 'dash'):
            for obj in prof.get(section, []):
                _scale_obj(obj, xoff, yoff)
        for section in ('menus', 'input'):
            for obj in prof.get(section, {}).values():
                _scale_obj(obj, xoff, yoff)
    return data


def main():
    with open(SRC) as f:
        src = json.load(f)
    out = scale_layout(src)
    with open(OUT, 'w') as f:
        json.dump(out, f, indent=2)
        f.write('\n')
    print(f'Wrote {OUT}')


if __name__ == '__main__':
    main()
```

import json
import os

BASE = os.path.abspath(os.path.join(os.path.dirname(__file__), '..'))


def _manifest():
    with open(os.path.join(BASE, 'wizard', 'wizard_manifest.json')) as f:
        return json.load(f)


def _config_default(entry, option_name):
    for opt in entry['config']:
        if opt['option_name'] == option_name:
            return opt['default']
    raise AssertionError(f'{option_name} not in config')


def test_manifest_entry_present():
    entry = _manifest()['modules']['display']['dsi_1024x768t']
    assert entry['filename'] == 'dsi_1024x768t'
    assert _config_default(entry, 'display_data_filename') == './display/dsi_1024x768t.json'
    assert entry['config'] != []


def test_default_display_config_includes_entry():
    # _default_display_config reads ./wizard/wizard_manifest.json relative to CWD.
    cwd = os.getcwd()
    os.chdir(BASE)
    try:
        from common.common import _default_display_config
        config = _default_display_config()
    finally:
        os.chdir(cwd)
    assert 'dsi_1024x768t' in config
    assert config['dsi_1024x768t']['display_data_filename'] == './display/dsi_1024x768t.json'
